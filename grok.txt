    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        ctx.imageSmoothingEnabled = false;
        canvas.focus();

        function resizeCanvas() {
            const rect = canvas.getBoundingClientRect();
            canvas.width = Math.floor(rect.width * window.devicePixelRatio);
            canvas.height = Math.floor(rect.height * window.devicePixelRatio);
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        let gameState = {
            player: { x: 434, y: 600, width: 32, height: 40, health: 3, speed: 8 },
            bullets: [], enemies: [], powerups: [], particles: [],
            score: 0, paused: false, keys: {}, lastShoot: 0,
            powerupsActive: {shield:0, rapidFire:0, multiShot:0, nuke:0, tripleShot:0}
        };

        const colors = {
            sky: '#87CEEB', ground: '#228B22',
            playerBody: '#2F4F4F', playerCockpit: '#87CEEB', playerWing: '#4682B4', 
            playerTail: '#1E90FF', playerEngine: '#FF4500'
        };

        // ‚úÖ 8 LO·∫†I M√ÅY BAY ƒê·ªäCH ƒê·∫∏P NH∆Ø TH·∫¨T
        const enemyTypes = [
            {body:'#C41E3A', cockpit:'#FF6B6B', wing:'#B22222', tail:'#8B0000', engine:'#FF4500'}, // F-16 Red
            {body:'#4169E1', cockpit:'#87CEEB', wing:'#1E90FF', tail:'#000080', engine:'#FF8C00'}, // F-15 Blue
            {body:'#32CD32', cockpit:'#90EE90', wing:'#228B22', tail:'#006400', engine:'#FFD700'}, // MiG Green
            {body:'#FFD700', cockpit:'#FFA500', wing:'#FF8C00', tail:'#DAA520', engine:'#FF4500'}, // Gold Bomber
            {body:'#9932CC', cockpit:'#DDA0DD', wing:'#8A2BE2', tail:'#9400D3', engine:'#FF1493'}, // Purple Jet
            {body:'#FF69B4', cockpit:'#FFB6C1', wing:'#FF1493', tail:'#C71585', engine:'#FF69B4'}, // Pink Fighter
            {body:'#696969', cockpit:'#A9A9A9', wing:'#808080', tail:'#2F4F4F', engine:'#DC143C'}, // Stealth Gray
            {body:'#8B4513', cockpit:'#DEB887', wing:'#A0522D', tail:'#654321', engine:'#FF6347'}  // Brown Attack
        ];

        // ‚úÖ PLAYER GI·ªêNG M√ÅY BAY TH·∫¨T (32x40) - View t·ª´ d∆∞·ªõi l√™n
        function drawPlayer(x, y) {
            ctx.save(); 
            ctx.translate(x + 16, y + 20);
            
            // C√°nh ch√≠nh (r·ªông, cong)
            ctx.fillStyle = colors.playerWing;
            ctx.fillRect(-18, -8, 36, 16);  // C√°nh ch√≠nh
            ctx.fillRect(-22, -4, 8, 8);    // C√°nh ngo√†i tr√°i
            ctx.fillRect(14, -4, 8, 8);     // C c√°nh ngo√†i ph·∫£i
            
            // Th√¢n m√°y bay (h√¨nh thu√¥n d√†i)
            ctx.fillStyle = colors.playerBody;
            ctx.fillRect(-10, -22, 20, 44); // Th√¢n ch√≠nh d√†i
            ctx.fillRect(-6, -28, 12, 10);  // M≈©i m√°y bay nh·ªçn
            
            // Cabin phi c√¥ng
            ctx.fillStyle = colors.playerCockpit;
            ctx.beginPath();
            ctx.arc(0, -18, 6, 0, Math.PI);
            ctx.fill();
            
            // ƒêu√¥i (2 c√°nh ƒë·ª©ng + ngang)
            ctx.fillStyle = colors.playerTail;
            // ƒêu√¥i ƒë·ª©ng tr√°i
            ctx.fillRect(-14, -14, 6, 20);
            ctx.fillRect(-16, -10, 4, 12);
            // ƒêu√¥i ƒë·ª©ng ph·∫£i  
            ctx.fillRect(8, -14, 6, 20);
            ctx.fillRect(12, -10, 4, 12);
            // ƒêu√¥i ngang
            ctx.fillRect(-12, -6, 24, 4);
            
            // ƒê·ªông c∆° (2 c√°i)
            ctx.fillStyle = '#000';
            ctx.fillRect(-8, 16, 6, 10);   // ƒê·ªông c∆° tr√°i
            ctx.fillRect(2, 16, 6, 10);    // ƒê·ªông c∆° ph·∫£i
            
            // L·ª≠a ƒë·ªông c∆°
            ctx.fillStyle = '#FFF';
            ctx.globalAlpha = 0.7;
            ctx.fillRect(-6, 18, 2, 4);
            ctx.fillRect(4, 18, 2, 4);
            ctx.globalAlpha = 1;
            ctx.fillStyle = colors.playerEngine;
            ctx.fillRect(-5, 24, 1, 8);    // L·ª≠a tr√°i
            ctx.fillRect(5, 24, 1, 8);     // L·ª≠a ph·∫£i
            
            // Khi√™n b·∫£o v·ªá
            if (gameState.powerupsActive.shield > 0) {
                ctx.strokeStyle = '#00FFFF';
                ctx.lineWidth = 3;
                ctx.shadowBlur = 25;
                ctx.shadowColor = '#00FFFF';
                ctx.beginPath();
                ctx.arc(0, 0, 28, 0, Math.PI*2);
                ctx.stroke();
                ctx.shadowBlur = 0;
            }
            
            ctx.restore();
        }

        // ‚úÖ ENEMY GI·ªêNG M√ÅY BAY TH·∫¨T (flip ƒë·ªÉ bay xu·ªëng)
        function drawEnemy(x, y, typeId) {
            const type = enemyTypes[typeId % enemyTypes.length];
            ctx.save(); 
            ctx.translate(x + 16, y + 20);
            ctx.scale(-1, 1); // Flip ƒë·ªÉ m≈©i h∆∞·ªõng xu·ªëng
            
            // C√°nh ch√≠nh
            ctx.fillStyle = type.wing;
            ctx.fillRect(-18, -8, 36, 16);
            ctx.fillRect(-22, -4, 8, 8);
            ctx.fillRect(14, -4, 8, 8);
            
            // Th√¢n m√°y bay
            ctx.fillStyle = type.body;
            ctx.fillRect(-10, -22, 20, 44);
            ctx.fillRect(-6, 12, 12, 10);  // M≈©i h∆∞·ªõng xu·ªëng
            
            // Cabin
            ctx.fillStyle = type.cockpit;
            ctx.beginPath();
            ctx.arc(0, -4, 6, 0, Math.PI);
            ctx.fill();
            
            // ƒêu√¥i
            ctx.fillStyle = type.tail;
            ctx.fillRect(-14, -14, 6, 20);
            ctx.fillRect(-16, -10, 4, 12);
            ctx.fillRect(8, -14, 6, 20);
            ctx.fillRect(12, -10, 4, 12);
            ctx.fillRect(-12, -6, 24, 4);
            
            // ƒê·ªông c∆°
            ctx.fillStyle = '#000';
            ctx.fillRect(-8, 16, 6, 10);
            ctx.fillRect(2, 16, 6, 10);
            
            // L·ª≠a ƒë·ªông c∆°
            ctx.fillStyle = '#FFF';
            ctx.globalAlpha = 0.6;
            ctx.fillRect(-6, 18, 2, 4);
            ctx.fillRect(4, 18, 2, 4);
            ctx.globalAlpha = 1;
            ctx.fillStyle = type.engine;
            ctx.fillRect(-5, 24, 1, 8);
            ctx.fillRect(5, 24, 1, 8);
            
            // Hi·ªáu ·ª©ng slow (xanh m·ªù)
            if (gameState.enemies.find(e => e.x === x && e.y === y)?.slowDown > 0) {
                ctx.fillStyle = 'rgba(0,255,0,0.2)';
                ctx.fillRect(-25, -35, 50, 70);
            }
            
            ctx.restore();
        }

        function drawBullet(b) {
            ctx.save(); 
            ctx.translate(b.x, b.y);
            ctx.fillStyle = '#FFFF00';
            ctx.fillRect(-5, -10, 10, 20);
            ctx.fillStyle = '#FFA500';
            ctx.fillRect(-3, 10, 6, 15);
            ctx.restore();
        }

        const powerupSprites = {bullet:'üíé', hp:'‚ù§Ô∏è', shield:'üõ°Ô∏è', rapid:'‚ö°', multi:'üî•', nuke:'üí£', triple:'‚≠ê'};
        function drawPowerup(x, y, type) {
            ctx.save(); 
            ctx.translate(x + 8, y + 8);
            ctx.font = 'bold 20px Arial'; 
            ctx.textAlign = 'center'; 
            ctx.textBaseline = 'middle';
            ctx.fillStyle = '#FFD700'; 
            ctx.shadowColor = '#FFF'; 
            ctx.shadowBlur = 15;
            ctx.fillText(powerupSprites[type], 0, 0); 
            ctx.shadowBlur = 0;
            ctx.restore();
        }

        // C√°c h√†m game gi·ªØ nguy√™n (spawn, update, etc...)
        function spawnEnemy() {
            gameState.enemies.push({
                x: 50 + Math.random() * (canvas.width/2 - 100),
                y: -50, speedX: (Math.random()-0.5)*1.0, speedY: 1.0 + Math.random()*0.6,
                type: Math.floor(Math.random()*8), width: 32, height: 40, health: 1, slowDown: 0,
                passed: false
            });
        }

        function spawnPowerup(x, y) {
            if (Math.random() < 0.18) {
                const types = ['bullet','hp','shield','rapid','multi','nuke','triple'];
                gameState.powerups.push({
                    x: x+Math.random()*20-10, y, type: types[Math.floor(Math.random()*types.length)],
                    speedY: 1.2, width: 16, height: 16
                });
            }
        }

        function tryShoot() {
            const now = Date.now();
            const cooldown = gameState.powerupsActive.rapidFire > 0 ? 70 : 130;
            if (now - gameState.lastShoot < cooldown) return;
            
            const mx = gameState.player.x + 16;
            const my = gameState.player.y;
            
            if (gameState.powerupsActive.tripleShot > 0) {
                gameState.bullets.push({x:mx, y:my, vx:0, vy:-10});
                gameState.bullets.push({x:mx-12, y:my, vx:-2.5, vy:-9});
                gameState.bullets.push({x:mx+12, y:my, vx:2.5, vy:-9});
            } else if (gameState.powerupsActive.multiShot > 0) {
                gameState.bullets.push({x:mx, y:my, vx:0, vy:-10});
                gameState.bullets.push({x:mx+10, y:my+6, vx:1.5, vy:-9});
                gameState.bullets.push({x:mx-10, y:my+6, vx:-1.5, vy:-9});
            } else {
                gameState.bullets.push({x:mx, y:my, vx:0, vy:-10});
            }
            gameState.lastShoot = now;
        }

        // Update v√† Render functions (gi·ªØ nguy√™n t·ª´ version tr∆∞·ªõc)
        function update() {
            if (gameState.paused) return;

            if (gameState.keys['ArrowLeft'] && gameState.player.x > 20) gameState.player.x -= gameState.player.speed;
            if (gameState.keys['ArrowRight'] && gameState.player.x < canvas.width/2 - 52) gameState.player.x += gameState.player.speed;

            if (gameState.keys['Space']) tryShoot();

            for (let i = gameState.bullets.length-1; i>=0; i--) {
                const b = gameState.bullets[i]; b.x += b.vx; b.y += b.vy;
                if (b.y < -30) gameState.bullets.splice(i,1);
            }

            for (let i = gameState.enemies.length-1; i>=0; i--) {
                const e = gameState.enemies[i];
                let speedMod = e.slowDown > 0 ? 0.25 : 1;
                e.x += e.speedX * speedMod; e.y += e.speedY * speedMod;
                if (e.slowDown > 0) e.slowDown--;

                if (e.x < 30 || e.x > canvas.width/2 - 62) e.speedX *= -1;
                
                if (e.y > canvas.height + 20) {
                    gameState.player.health--;
                    createExplosion(e.x+16, canvas.height);
                    gameState.enemies.splice(i,1);
                    updateUI();
                    if (gameState.player.health <= 0) {
                        alert('GAME OVER! Final Score: ' + gameState.score);
                        resetGame();
                    }
                    continue;
                }

                for (let j = gameState.bullets.length-1; j>=0; j--) {
                    const b = gameState.bullets[j];
                    if (Math.abs(b.x-(e.x+16)) < 18 && Math.abs(b.y-(e.y+20)) < 20) {
                        gameState.bullets.splice(j,1); e.slowDown = 150; e.health--;
                        createExplosion(e.x+16, e.y+20);
                        if (e.health <= 0) {
                            gameState.score += 15 + e.type*3; spawnPowerup(e.x, e.y);
                            gameState.enemies.splice(i,1); updateUI();
                        }
                        break;
                    }
                }

                if (!gameState.powerupsActive.shield && Math.abs(e.x-gameState.player.x)<28 && Math.abs(e.y-gameState.player.y)<30) {
                    gameState.player.health--; createExplosion(e.x+16, e.y+20);
                    gameState.enemies.splice(i,1); updateUI();
                    if (gameState.player.health <= 0) {
                        alert('GAME OVER! Final Score: ' + gameState.score);
                        resetGame();
                    }
                }
            }

            for (let i = gameState.powerups.length-1; i>=0; i--) {
                const p = gameState.powerups[i]; p.y += p.speedY;
                if (p.y > canvas.height) gameState.powerups.splice(i,1);
                if (Math.abs(p.x-gameState.player.x)<24 && Math.abs(p.y-gameState.player.y)<24) {
                    activatePowerup(p.type); gameState.powerups.splice(i,1); createCollectEffect(p.x,p.y);
                }
            }

            Object.keys(gameState.powerupsActive).forEach(k => {
                if (gameState.powerupsActive[k] > 0) gameState.powerupsActive[k]--;
            });
            if (gameState.powerupsActive.nuke > 0) {
                gameState.enemies.forEach(e => createExplosion(e.x+16,e.y+20));
                gameState.enemies = []; gameState.powerupsActive.nuke = 0; gameState.score += 150;
            }
            if (Math.random() < 0.01) spawnEnemy();

            for (let i = gameState.particles.length-1; i>=0; i--) {
                const p = gameState.particles[i]; p.x += p.vx; p.y += p.vy; p.life--;
                if (p.life <= 0) gameState.particles.splice(i,1);
            }
        }

        function activatePowerup(type) {
            switch(type) {
                case 'bullet': case 'hp': gameState.player.health = Math.min(6, gameState.player.health + 1); break;
                case 'shield': gameState.powerupsActive.shield = 360; break;
                case 'rapid': gameState.powerupsActive.rapidFire = 720; break;
                case 'multi': gameState.powerupsActive.multiShot = 540; break;
                case 'nuke': gameState.powerupsActive.nuke = 1; break;
                case 'triple': gameState.powerupsActive.tripleShot = 1080; break;
            }
            updateUI();
        }

        function createExplosion(x, y) {
            for (let i = 0; i < 25; i++) gameState.particles.push({
                x, y, vx:(Math.random()-0.5)*12, vy:(Math.random()-0.5)*12, life:40+Math.random()*25
            });
        }
        function createCollectEffect(x, y) {
            for (let i = 0; i < 12; i++) gameState.particles.push({
                x, y, vx:(Math.random()-0.5)*8, vy:(Math.random()-0.5)*8, life:50, color:'#FFD700'
            });
        }

        function render() {
            const w = canvas.width, h = canvas.height;
            const gradient = ctx.createLinearGradient(0,0,0,h);
            gradient.addColorStop(0, colors.sky); gradient.addColorStop(0.6,'#98D8E8'); gradient.addColorStop(1,colors.ground);
            ctx.fillStyle = gradient; ctx.fillRect(0,0,w,h);

            ctx.fillStyle = '#32CD32'; ctx.fillRect(0, h*0.75, w, h*0.25);
            ctx.fillStyle = '#228B22'; ctx.fillRect(0, h*0.85, w, h*0.15);

            drawPlayer(gameState.player.x, gameState.player.y);
            gameState.bullets.forEach(drawBullet);
            gameState.enemies.forEach((e,i) => drawEnemy(e.x, e.y, e.type));
            gameState.powerups.forEach(p => drawPowerup(p.x, p.y, p.type));

            gameState.particles.forEach(p => {
                ctx.fillStyle = p.color || '#FFAA00'; ctx.globalAlpha = p.life/60;
                ctx.fillRect(p.x, p.y, 5, 5);
            }); ctx.globalAlpha = 1;

            if (gameState.paused) {
                ctx.fillStyle = 'rgba(0,0,0,0.8)'; ctx.fillRect(0,0,w,h);
                ctx.fillStyle = '#FFF'; ctx.font = 'bold 64px monospace'; ctx.textAlign = 'center';
                ctx.fillText('PAUSED', w/2, h/2); ctx.textAlign = 'left';
            }
        }

        function updateUI() {
            document.getElementById('score').textContent = gameState.score;
            document.getElementById('health').textContent = gameState.player.health;
            const pw = document.getElementById('powerups');
            let active = [];
            if (gameState.powerupsActive.shield > 0) active.push('üõ°Ô∏è');
            if (gameState.powerupsActive.rapidFire > 0) active.push('‚ö°');
            if (gameState.powerupsActive.multiShot > 0) active.push('üî•');
            if (gameState.powerupsActive.tripleShot > 0) active.push('‚≠ê');
            pw.innerHTML = active.length ? active.join(' ') : 'Click canvas ‚Üí SPACE b·∫Øn!';
        }

        window.addEventListener('keydown', e => {
            gameState.keys[e.code] = true; if (e.code === 'Space') e.preventDefault();
        });
        window.addEventListener('keyup', e => gameState.keys[e.code] = false);
        canvas.addEventListener('click', () => canvas.focus());

        window.togglePause = () => gameState.paused = !gameState.paused;
        window.resetGame = () => {
            gameState = {
                ...gameState, player: {x:434, y:600, width:32, height:40, health:3, speed:8},
                bullets:[], enemies:[], powerups:[], particles:[], score:0,
                powerupsActive:{shield:0,rapidFire:0,multiShot:0,nuke:0,tripleShot:0}, lastShoot:0
            };
            updateUI();
        };

        let lastTime = 0;
        function gameLoop(time) {
            if (time - lastTime > 1000/60) { update(); render(); lastTime = time; }
            requestAnimationFrame(gameLoop);
        }

        updateUI(); requestAnimationFrame(gameLoop);
    </script>
